const SHEET_NAME = "main";
const FOLDER_NAME = "files"; 
const VERSION = "v0.03";

/**
 * [步驟 A] 初始化資料庫與權限
 * 請在部署「新部署」前執行一次，完成 Drive 權限核准
 */
function initDatabase() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // 1. 初始化 main 工作表
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
  }
  const header = ["row_type", "user_id", "pin", "reward_code", "content"];
  sheet.getRange(1, 1, 1, header.length).setValues([header]);
  sheet.setFrozenRows(1);
  sheet.getRange(1, 1, 1, header.length).setFontWeight("bold").setBackground("#f3f4f6");

  // 2. 觸發權限：建立/檢查資料夾
  try {
    const folder = getOrCreateFilesFolder();
    console.log("初始化成功！檔案將儲存於: " + folder.getName());
    console.log("資料夾連結: " + folder.getUrl());
  } catch(e) {
    throw new Error("Drive 存取授權失敗: " + e.message);
  }
}

/**
 * [步驟 B] 處理前端 POST 請求
 */
function doPost(e) {
  try {
    if (!e || !e.postData || !e.postData.contents) {
      return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": "No content"})).setMimeType(ContentService.MimeType.JSON);
    }

    const payload = JSON.parse(e.postData.contents);
    let fileUrl = "";

    // 處理檔案上傳
    if (payload.file_data && payload.file_name) {
      try {
        fileUrl = saveFileToDrive(payload.file_data, payload.file_name, payload.user_id);
      } catch (fileErr) {
        fileUrl = "Upload Failed: " + fileErr.message;
      }
    }
    
    payload.file_url = fileUrl;
    appendSurveyRows(payload);
    
    return ContentService.createTextOutput(JSON.stringify({
      "status": "success",
      "user_id": payload.user_id
    })).setMimeType(ContentService.MimeType.JSON);
    
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      "status": "error", 
      "message": error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * 結構化寫入 Sheet
 */
function appendSurveyRows(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME) || getOrCreateSheet(SHEET_NAME);
  const rows = [];
  const common = [p.user_id, p.pin, p.reward_code];

  // Meta
  rows.push(["meta", ...common, `timestamp=${p.timestamp};;;version=${VERSION}`]);

  // Q1-Q4
  const qConfigs = [
    { id: "q1", qid: "Q1", text: "你目前對AI發展與影響的感受最接近哪些？" },
    { id: "q2", qid: "Q2", text: "你目前在工作或學習中使用AI的情況最接近哪些？" },
    { id: "q3", qid: "Q3", text: "如果要在工作或組織中導入AI，你目前遇到的困難有哪些？" },
    { id: "q4", qid: "Q4", text: "如果有一個AI平台，你最希望它提供哪些內容或幫助？" }
  ];

  qConfigs.forEach(conf => {
    const ans = (p[conf.id] || []).join("||");
    const other = (p[conf.id + "_other"] || "").replace(/;;;/g, " ");
    const content = `qid=${conf.qid};;;question=${conf.text};;;ans=${ans};;;other=${other}`;
    rows.push([conf.id, ...common, content]);
  });

  // Q5
  const q5Content = `qid=Q5;;;question=如果你有更具體的需求或想法;;;text=${(p.q5_text || "").replace(/;;;/g, " ")};;;contact=${(p.contact || "")};;;file=${p.file_url || ""}`;
  rows.push(["q5", ...common, q5Content]);

  sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, rows[0].length).setValues(rows);
}

/**
 * 儲存 Base64 檔案
 */
function saveFileToDrive(base64Full, fileName, userId) {
  const folder = getOrCreateFilesFolder();
  const parts = base64Full.split(",");
  const body = parts[1];
  const contentType = parts[0].split(":")[1].split(";")[0];
  
  const blob = Utilities.newBlob(Utilities.base64Decode(body), contentType, `${userId}_${fileName}`);
  const file = folder.createFile(blob);
  
  return file.getUrl();
}

/**
 * 取得 files 資料夾（優化版）
 */
function getOrCreateFilesFolder() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const fileId = ss.getId();
  const parents = DriveApp.getFileById(fileId).getParents();
  
  let parentFolder;
  if (parents.hasNext()) {
    parentFolder = parents.next();
  } else {
    parentFolder = DriveApp.getRootFolder();
  }
  
  const folders = parentFolder.getFoldersByName(FOLDER_NAME);
  if (folders.hasNext()) {
    return folders.next();
  } else {
    return parentFolder.createFolder(FOLDER_NAME);
  }
}

function getOrCreateSheet(name) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.getRange(1, 1, 1, 5).setValues([["row_type", "user_id", "pin", "reward_code", "content"]]);
  }
  return sheet;
}
